name: Daily Build and Test
on:
  schedule:
    - cron: '30 10 * * *'
  workflow_dispatch:
jobs:
  build-and-test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-python@v4
        with:
          python-version: '3.x'
      - name: Build and test
        run: |
          echo 'Starting daily build...'
          python run_tests.py
          echo 'Daily build completed!'
EOF
