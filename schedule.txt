name: Midnight Task
on:
  schedule:
    - cron: '0 0 * * *'
  workflow_dispatch:
jobs:
  run-midnight-task:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-python@v4
        with:
          python-version: '3.x'
      - name: Execute midnight task
        run: python daily_task.py
EOF
