name: Build
on: [push]
jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Show commit info
        run: |
          echo 'Latest commit:'
          git log -1 --pretty=format:'%h - %s (%an)'
      - name: Build success
        run: echo 'Build completed successfully!'
